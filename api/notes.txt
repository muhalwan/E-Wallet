# To create. Only admin can access
POST /api/create.php

# To get all users. Only admin can access
GET /api/read.php

# To get particular user. Every user can access his data only. Admin can access all
GET /api/read.php/<id>

# To update particular user. Every user can update his data only. Admin can update all
PUT /api/update.php/<id>

# To delete particular user. Only admin can access
DELETE /api/delete.php/<id>
